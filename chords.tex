  \expandafter\newcommand\csname chord#4\endcsname{%
    \IfNoValueTF{#1}%
      {\chordbox{#5}{#3}}
      {\bchordbox[#1]{#5}{#3}{\IfNoValueTF{#2}{#1}{#2}}}%
    }%

  \expandafter\NewDocumentCommand\csname tabchord#4\endcsname{oO{#3}}{%
    \IfNoValueTF{##1}{\barcomment{#5}}{\barcomment[##1]{#5}}%
    \setchord{##2}
  }
}
